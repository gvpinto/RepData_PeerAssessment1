temp = spread(origActivity, interval, steps)
names = colnames(temp)
for (name in names) {
    temp[, name] = impute(temp[, name], mean)
}

imputedActivity = gather(temp, interval, steps, -date)
imputedActivity$interval = as.numeric(as.character(imputedActivity$interval))
